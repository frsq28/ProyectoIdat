
    Definir carrito[10], cantidades[10] Como Entero
    Definir total, precioProducto Como Real
    Definir i Como Entero


    carrito[1] <- 101 // Camiseta
    cantidades[1] <- 2
    carrito[2] <- 102 // Pantalón
    cantidades[2] <- 1
    carrito[3] <- 0 // Fin del carrito

    total <- 0

    Para i <- 1 Hasta 10 Con Paso 1
        Si carrito[i] <> 0 Entonces
            // Asignar precio según producto
            Segun carrito[i] Hacer
                101:
                    precioProducto <- 40 // Camiseta
                102:
                    precioProducto <- 60 // Pantalón
                103:
                    precioProducto <- 100 // Casaca
                De Otro Modo:
                    precioProducto <- 0
            FinSegun

            total <- total + (precioProducto * cantidades[i])
        FinSi
    FinPara

    Si total = 0 Entonces
        Escribir "El carrito está vacío. No se puede crear pedido."
    Sino
        Escribir "Pedido creado con éxito."
        Escribir "Total a pagar: S/.", total
    FinSi
FinProceso
